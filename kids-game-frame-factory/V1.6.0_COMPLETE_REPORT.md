## ✅ PauseManagerComponent 详解

### 核心功能

#### 1. 暂停状态机 ⭐⭐⭐⭐⭐

```typescript
enum PauseState {
  PLAYING = 'playing',    // 游戏中（未暂停）
  PAUSED = 'paused',      // 已暂停
  PAUSING = 'pausing',    // 暂停中（过渡状态）
  RESUMING = 'resuming'   // 恢复中（过渡状态）
}

// 状态流转
PLAYING → PAUSING → PAUSED → RESUMING → PLAYING
```

**使用示例**:
```typescript
const pauseManager = new PauseManagerComponent(scene)
pauseManager.init({
  showOverlay: true,
  overlayColor: 0x000000,
  overlayAlpha: 0.5,
  stopAllSounds: true,
  disableInput: true,
  transitionDuration: 300,
  enableKeyboardShortcut: true
})

// 检查状态
if (pauseManager.isPaused()) {
  console.log('游戏已暂停')
}

console.log(pauseManager.getState())  // 'paused'
```

#### 4. 音效和输入控制 ⭐⭐⭐⭐⭐

```typescript
pause(): void {
  // ✅ 停止所有音效
  if (stopAllSounds) {

  }
  
  // ✅ 禁用输入
  if (disableInput) {
    scene.input.enabled = false
  }

resume(): void {
  // ✅ 恢复所有音效
  if (stopAllSounds) {

  }
  
  // ✅ 启用输入

  }
}
```

- 暂停时：背景音乐、音效全部静音

- 暂停时：键盘、鼠标、触摸输入全部禁用
- 恢复时：输入系统重新激活

```typescript
import { PauseManagerComponent } from 'kids-game-frame-factory'

class GameScene extends Phaser.Scene {
  private pauseManager: PauseManagerComponent
  
  create() {
    this.pauseManager = new PauseManagerComponent(this)
    
    this.pauseManager.init({
      showOverlay: true,
      overlayColor: 0x000000,
      overlayAlpha: 0.5,
      stopAllSounds: true,
      disableInput: true,
      transitionDuration: 300,
      enableKeyboardShortcut: true
    })
    
    // 监听暂停事件
    EventBus.getInstance().on(
      GameEventType.PAUSE,
      (event) => {
        console.log(`⏸️ 游戏暂停：${event.payload.reason}`)
        playPauseSound()
      }
    )
    
    // 监听恢复事件
    EventBus.getInstance().on(
      GameEventType.RESUME,
      (event) => {
        console.log(`▶️ 游戏恢复，暂停了 ${event.payload.pauseDuration}ms`)
        playResumeSound()
      }
    )
  }
  
  update(time: number, delta: number) {
    this.pauseManager.update(delta)
  }
}
```

#### 自定义暂停菜单

```typescript
class GameScene extends Phaser.Scene {
  private pauseManager: PauseManagerComponent
  private pauseMenu: Phaser.GameObjects.Container | null = null
  
  create() {
    this.pauseManager = new PauseManagerComponent(this)
    
    this.pauseManager.init({
      showOverlay: true,
      enableKeyboardShortcut: false  // 禁用默认快捷键
    })
    
    // 创建自定义暂停菜单
    this.createPauseMenu()
    
    // 监听暂停事件
    EventBus.getInstance().on(
      GameEventType.PAUSE,
      () => {
        this.showPauseMenu()
      }
    )
    
    // 监听恢复事件
    EventBus.getInstance().on(
      GameEventType.RESUME,
      () => {
        this.hidePauseMenu()
      }
    )
  }
  
  private createPauseMenu(): void {
    const width = this.scale.width
    const height = this.scale.height
    
    this.pauseMenu = this.add.container(width / 2, height / 2)
    
    // 菜单背景
    const bg = this.add.rectangle(0, 0, 400, 300, 0x333333, 0.9)
    bg.setStrokeStyle(2, 0xffffff)
    
    // 标题
    const title = this.add.text(0, -100, 'PAUSED', {
      fontFamily: 'Arial',
      fontSize: '48px',
      color: '#ffffff'
    })
    title.setOrigin(0.5)
    
    // 继续按钮
    const resumeBtn = this.add.text(0, 0, '继续游戏', {
      fontFamily: 'Arial',
      fontSize: '24px',
      color: '#ffffff',
      backgroundColor: '#4CAF50',
      padding: { x: 20, y: 10 }
    })
    resumeBtn.setOrigin(0.5)
    resumeBtn.setInteractive({ useHandCursor: true })
    resumeBtn.on('pointerdown', () => {
      this.pauseManager.resume()
    })
    
    // 重新开始按钮
    const restartBtn = this.add.text(0, 60, '重新开始', {
      fontFamily: 'Arial',
      fontSize: '24px',
      color: '#ffffff',
      backgroundColor: '#2196F3',
      padding: { x: 20, y: 10 }
    })
    restartBtn.setOrigin(0.5)
    restartBtn.setInteractive({ useHandCursor: true })
    restartBtn.on('pointerdown', () => {
      this.scene.restart()
    })
    
    // 返回主菜单按钮
    const menuBtn = this.add.text(0, 120, '返回主菜单', {
      fontFamily: 'Arial',
      fontSize: '24px',
      color: '#ffffff',
      backgroundColor: '#FF5722',
      padding: { x: 20, y: 10 }
    })
    menuBtn.setOrigin(0.5)
    menuBtn.setInteractive({ useHandCursor: true })
    menuBtn.on('pointerdown', () => {
      this.scene.scene.start('MainMenu')
    })
    
    this.pauseMenu.add([bg, title, resumeBtn, restartBtn, menuBtn])
    this.pauseMenu.setVisible(false)
  }
  
  private showPauseMenu(): void {
    this.pauseMenu?.setVisible(true)
  }
  
  private hidePauseMenu(): void {
    this.pauseMenu?.setVisible(false)
  }
}
```

#### 自动暂停场景

```typescript
class GameScene extends Phaser.Scene {
  private pauseManager: PauseManagerComponent
  
  create() {
    this.pauseManager = new PauseManagerComponent(this)
    this.pauseManager.init({})
    
    // 失去焦点时自动暂停
    window.addEventListener('blur', () => {
      if (!this.pauseManager.isPaused()) {
        this.pauseManager.pause('window_blur')
      }
    })
    
    // 获得焦点时不自动恢复（需要用户手动操作）
    // 避免玩家在没准备好时突然开始游戏
    
    // 游戏结束时自动暂停
    EventBus.getInstance().on(
      GameEventType.GAME_OVER,
      () => {
        this.pauseManager.pause('game_over')
      }
    )
    
    // 显示重要消息时自动暂停
    EventBus.getInstance().on(
      GameEventType.SHOW_MESSAGE,
      (event) => {
        if (event.payload.priority === 'high') {
          this.pauseManager.pause('show_message')
        }
      }
    )
  }
}
```

### 完整的游戏组装

```typescript
import { 
  GameStateComponent,
  GridMovementComponent,
  ScoreManagerComponent,
  InputHandlerComponent,
  CollisionDetectionComponent,
  ItemSpawnerComponent,
  PauseManagerComponent
} from 'kids-game-frame-factory'

class CompleteGameScene extends Phaser.Scene {
  create() {
    // 创建所有组件
    const gameState = new GameStateComponent(this)
    const gridMovement = new GridMovementComponent(this)
    const scoreManager = new ScoreManagerComponent(this)
    const inputHandler = new InputHandlerComponent(this)
    const collision = new CollisionDetectionComponent(this)
    const itemSpawner = new ItemSpawnerComponent(this)
    const pauseManager = new PauseManagerComponent(this)
    
    // 初始化
    gameState.init({})
    gridMovement.init({ speed: 200, gridCols: 32, gridRows: 18, cellSize: 40 })
    scoreManager.init({ enableCombo: true })
    inputHandler.init({ enableKeyboard: true })
    collision.init({ debugMode: false })
    itemSpawner.init({ maxItems: 5 })
    pauseManager.init({ enableKeyboardShortcut: true })
    
    // 监听暂停事件
    EventBus.getInstance().on(
      GameEventType.PAUSE,
      () => {
        console.log('⏸️ 游戏暂停')
        // 显示暂停菜单、播放音效等
      }
    )
    
    // 监听恢复事件
    EventBus.getInstance().on(
      GameEventType.RESUME,
      () => {
        console.log('▶️ 游戏恢复')
        // 隐藏暂停菜单等
      }
    )
    
    // 开始游戏
    gameState.startGame()
  }
  
  update(time: number, delta: number) {
    // 只在非暂停状态下更新
    if (!pauseManager.isPaused()) {
      gridMovement.update(delta)
      scoreManager.update(delta)
      inputHandler.update(delta)
      collision.update(delta)
      itemSpawner.update(delta)
    }
    pauseManager.update(delta)
  }
}
```
