#### 3. 特效管理系统 ⭐⭐⭐⭐⭐

```typescript
class ParticleRenderer {
  // ⭐ 活跃的特效列表
  private activeEffects: Map<string, ActiveParticleEffect> = new Map()
  
  // ⭐ 创建特效
  createEffect(config): string {
    const id = `effect_${++counter}`
    const emitter = createEmitter(config)
    
    activeEffects.set(id, {
      id,
      type: 'custom',
      emitter,
      createdAt: Date.now(),
      duration: config.lifespan ?? 1000
    })
    
    return id  // 返回 ID 用于后续控制
  }
  
  // ⭐ 播放预设特效
  playPresetEffect(preset): string {
    const config = getPresetConfig(preset.type)
    config.x = preset.position.x
    config.y = preset.position.y
    
    return createEffect(config)
  }
  
  // ⭐ 停止特效
  stopEffect(id): void {
    const effect = activeEffects.get(id)
    if (effect) {
      effect.emitter.stop()
    }
  }
  
  // ⭐ 销毁特效
  destroyEffect(id): void {
    const effect = activeEffects.get(id)
    if (effect) {
      effect.emitter.destroy()
      activeEffects.delete(id)
    }
  }
  
  // ⭐ 自动清理过期特效
  update(_deltaTime): void {
    const now = Date.now()
    
    activeEffects.forEach((effect, id) => {
      const age = now - effect.createdAt
      
      if (age >= effect.duration) {
        // 到期，销毁
        if (effect.onComplete) effect.onComplete()
        destroyEffect(id)
      }
    })
    
    // 限制最大数量
    if (activeEffects.size > maxActiveEffects) {
      const oldestId = Array.from(keys)[0]
      destroyEffect(oldestId)
    }
  }
}
```

#### 4. 事件驱动自动播放 ⭐⭐⭐⭐⭐

```typescript
handleEvent(event): void {
  switch (event.type) {
    case GameEventType.COLLISION_DETECTED:
      // 碰撞时自动播放爆炸特效
      if (event.payload.eventType === 'start') {
        const pos = event.payload.contactPoint
        if (pos) {
          playPresetEffect({
            type: 'explosion',
            position: pos
          })
        }
      }
      break
      
    case GameEventType.ITEM_COLLECTED:
      // 物品被收集时自动播放收集特效
      const itemPos = event.payload.position
      if (itemPos) {
        playPresetEffect({
          type: 'collection',
          position: itemPos
        })
      }
      break
      
    case GameEventType.GAME_OVER:
      // 游戏结束时播放彩带庆祝
      playPresetEffect({
        type: 'confetti',
        position: { 
          x: screenWidth / 2, 
          y: screenHeight / 2 
        }
      })
      break
  }
}
```

**效果**:

```typescript
import { ParticleRenderer } from 'kids-game-frame-factory'

class SnakeGameScene extends Phaser.Scene {
  private particleRenderer: ParticleRenderer
  
  create() {
    this.particleRenderer = new ParticleRenderer(this)
    
    this.particleRenderer.init({
      defaultCount: 30,
      defaultLifespan: 1000,
      maxActiveEffects: 50,
      debugMode: false
    })
    
    // 监听碰撞事件（自动播放爆炸特效）
    EventBus.getInstance().on(
      GameEventType.COLLISION_DETECTED,
      (event) => {
        // 不需要手动播放，ParticleRenderer 会自动处理
        console.log('💥 碰撞发生，自动播放爆炸特效')
      }
    )
    
    // 监听食物收集事件（自动播放收集特效）
    EventBus.getInstance().on(
      GameEventType.ITEM_COLLECTED,
      (event) => {
        console.log('✨ 食物被收集，自动播放收集特效')
      }
    )
  }
  
  update(deltaTime: number) {
    this.particleRenderer.update(deltaTime)
    
    // 获取统计信息
    const stats = this.particleRenderer.getStats()
    console.log(`活跃特效数：${stats.activeEffects}`)
  }
}
```

```typescript
class SpaceShooterScene extends Phaser.Scene {
  private particleRenderer: ParticleRenderer
  
  create() {
    this.particleRenderer = new ParticleRenderer(this)
    
    this.particleRenderer.init({
      defaultCount: 30,
      defaultLifespan: 1000,
      maxActiveEffects: 200  // 大量特效
    })
    
    // 玩家发射子弹
    this.events.on('player_shoot', (position) => {
      // 枪口火焰
      this.particleRenderer.playPresetEffect({
        type: 'fire',
        position: { x: position.x, y: position.y - 20 }
      })
    })
    
    // 敌人被击中
    EventBus.getInstance().on(
      GameEventType.COLLISION_DETECTED,
      (event) => {
        if (event.payload.objectB?.type === 'enemy') {
          const pos = event.payload.contactPoint
          
          // 爆炸特效
          this.particleRenderer.playPresetEffect({
            type: 'explosion',
            position: pos
          })
          
          // 火花特效
          this.particleRenderer.playPresetEffect({
            type: 'sparkle',
            position: pos
          })
        }
      }
    )
    
    // 玩家死亡
    this.events.on('player_death', (position) => {
      // 大爆炸
      for (let i = 0; i < 5; i++) {
        setTimeout(() => {
          this.particleRenderer.playPresetEffect({
            type: 'explosion',
            position: {
              x: position.x + (Math.random() - 0.5) * 100,
              y: position.y + (Math.random() - 0.5) * 100
            }
          })
        }, i * 100)
      }
    })
    
    // Boss 战胜利
    this.events.on('boss_defeated', (position) => {
      // 盛大的彩带雨
      for (let i = 0; i < 10; i++) {
        setTimeout(() => {
          this.particleRenderer.playPresetEffect({
            type: 'confetti',
            position: {
              x: Math.random() * screenWidth,
              y: 0
            }
          })
        }, i * 200)
      }
    })
  }
  
  update(deltaTime: number) {
    this.particleRenderer.update(deltaTime)
    
    // 监控性能
    const stats = this.particleRenderer.getStats()
    if (stats.activeEffects > 150) {
      console.warn('⚠️ 特效过多，可能影响性能')
    }
  }
}
```
