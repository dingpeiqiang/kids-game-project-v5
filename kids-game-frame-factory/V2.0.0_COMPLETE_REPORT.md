#### 3. 对象池系统 ⭐⭐⭐⭐⭐

```typescript
// ⭐ 初始化对象池
initializeObjectPools(): void {
  const types: GameObjectType[] = [
    'snake', 'food', 'obstacle', 
    'player', 'enemy', 'item', 'projectile'
  ]
  
  types.forEach(type => {
    const pool = scene.add.group({
      classType: Phaser.GameObjects.Container,
      maxSize: poolSize ?? 100,
      runChildUpdate: false
    })
    
    objectPools.set(type, pool)
  })
}

// ⭐ 从池中获取对象
createObject(config): GameObject {
  if (enableObjectPool && objectPools.has(config.type)) {
    const pool = objectPools.get(config.type)!
    const pooledObject = pool.getFirstDead(false)
    
    if (pooledObject) {
      // 复用池中的对象
      gameObject = pooledObject
      gameObject.setActive(true)
      gameObject.setVisible(true)
    } else {
      // 池中无可用对象，创建新的
      gameObject = createGameObject(config)
      pool.add(gameObject)
    }
  } else {
    gameObject = createGameObject(config)
  }
  
  gameObjects.set(config.id, gameObject)
  return gameObject
}

// ⭐ 回收到池中
destroyObject(id): void {
  const gameObject = gameObjects.get(id)
  const config = objectConfigs.get(id)
  
  if (enableObjectPool && config && objectPools.has(config.type)) {
    const pool = objectPools.get(config.type)!
    
    if (pool.contains(gameObject)) {
      // 回收到池中而非销毁
      gameObject.setActive(false)
      gameObject.setVisible(false)
    } else {
      gameObject.destroy()
    }
  } else {
    gameObject.destroy()
  }
  
  gameObjects.delete(id)
  objectConfigs.delete(id)
}
```

**性能优势**:
```
不使用对象池:
创建 100 个食物 → 100 次内存分配
销毁 100 个食物 → 100 次垃圾回收

使用对象池:
创建 100 个食物 → 最多 100 次分配（可复用）
销毁 100 个食物 → 0 次垃圾回收（回收到池中）
```

#### 4. 平滑移动插值 ⭐⭐⭐⭐⭐

```typescript
updatePosition(id, position, smooth = false): void {
  const gameObject = gameObjects.get(id)
  const config = objectConfigs.get(id)
  
  if (!gameObject || !config) return
  
  if (smooth && enableAnimation) {
    // Tween 平滑移动（100ms）
    scene.tweens.add({
      targets: gameObject,
      x: position.x,
      y: position.y,
      duration: 100,
      ease: 'Linear'
    })
  } else {
    // 立即移动
    ;(gameObject as any).x = position.x
    ;(gameObject as any).y = position.y
  }
  
  // 更新配置
  config.x = position.x
  config.y = position.y
  objectConfigs.set(id, config)
}
```

**效果对比**:
```
立即移动:
帧 1: (200, 150)
帧 2: (240, 150)  ← 瞬移

平滑移动:
帧 1: (200, 150)
帧 1.5: (220, 150)  ← 中间状态
帧 2: (240, 150)
```

```typescript
// ⭐ 播放动画
playAnimation(id, key, repeat = -1): void {
  const gameObject = gameObjects.get(id)
  
  if (!gameObject || !enableAnimation) return
  
  if ('play' in gameObject) {
    ;(gameObject as any).play({ key, repeat })
  }
}

// ⭐ 停止动画
stopAnimation(id): void {
  const gameObject = gameObjects.get(id)
  
  if (!gameObject) return
  
  if ('stopPlayback' in gameObject) {
    ;(gameObject as any).stopPlayback()
  }
}

// 使用示例
gameObjectRenderer.playAnimation('player', 'walk_cycle', -1)  // 无限循环
gameObjectRenderer.playAnimation('enemy', 'attack', 0)        // 播放一次
gameObjectRenderer.stopAnimation('player')                    // 停止
```

#### 贪吃蛇游戏

```typescript
import { GameObjectRenderer } from 'kids-game-frame-factory'

class SnakeGameScene extends Phaser.Scene {
  private gameObjectRenderer: GameObjectRenderer
  
  create() {
    this.gameObjectRenderer = new GameObjectRenderer(this)
    
    this.gameObjectRenderer.init({
      enableObjectPool: true,
      poolSize: 100,
      enableAnimation: true,
      autoManageDepth: true
    })
    
    // 创建蛇头
    this.gameObjectRenderer.createObject({
      id: 'snake_head',
      type: 'snake',
      x: 200,
      y: 150,
      width: 40,
      height: 40,
      color: 0x00ff00,
      depth: 30
    })
    
    // 创建蛇身
    for (let i = 1; i < 5; i++) {
      this.gameObjectRenderer.createObject({
        id: `snake_body_${i}`,
        type: 'snake',
        x: 200 - i * 40,
        y: 150,
        width: 40,
        height: 40,
        color: 0x00cc00,
        depth: 30
      })
    }
    
    // 创建食物
    this.gameObjectRenderer.createObject({
      id: 'food_1',
      type: 'food',
      x: 400,
      y: 300,
      radius: 15,
      color: 0xff0000,
      depth: 0
    })
    
    // 监听碰撞事件
    EventBus.getInstance().on(
      GameEventType.COLLISION_DETECTED,
      (event) => {
        if (event.payload.objectB === 'food') {
          // 食物被吃掉，销毁并重新生成
          this.gameObjectRenderer.destroyObject(event.payload.itemId)
          this.spawnFood()
        }
      }
    )
  }
  
  update(deltaTime: number) {
    this.gameObjectRenderer.update(deltaTime)
    
    // 更新蛇头位置（平滑移动）
    const newHeadPos = this.calculateNewHeadPosition()
    this.gameObjectRenderer.updatePosition('snake_head', newHeadPos, true)
    
    // 更新蛇身位置（跟随蛇头）
    this.updateSnakeBody()
  }
  
  private spawnFood(): void {
    const x = Math.random() * 800
    const y = Math.random() * 600
    
    this.gameObjectRenderer.createObject({
      id: 'food_new',
      type: 'food',
      x,
      y,
      radius: 15,
      color: 0xff0000
    })
  }
}
```

```typescript
    for (let i = 0; i < 10; i++) {
      this.gameObjectRenderer.createObject({
        id: `coin_${i}`,
        type: 'item',
        x: 100 + i * 100,
        y: 300,
        width: 20,
        height: 20,
        textureKey: 'coin_sprite',
        depth: 10
      })
      
      this.gameObjectRenderer.playAnimation(`coin_${i}`, 'spin', -1)
    }
  }
  
  update(deltaTime: number) {
    this.gameObjectRenderer.update(deltaTime)
    
    // 更新玩家位置
    const playerPos = this.inputManager.getPlayerPosition()
    this.gameObjectRenderer.updatePosition('player', playerPos, false)
    
    // 更新敌人 AI
    this.updateEnemiesAI()
  }
  
  private updateEnemiesAI(): void {
    // 简单的巡逻 AI
    for (let i = 0; i < 5; i++) {
      const enemy = this.gameObjectRenderer.getObject(`enemy_${i}`)
      const config = this.gameObjectRenderer.getConfig(`enemy_${i}`)
      
      if (config && enemy) {
        const newX = config.x + Math.sin(Date.now() / 1000) * 2
        this.gameObjectRenderer.updatePosition(`enemy_${i}`, { 
          x: newX, 
          y: config.y 
        }, true)
      }
    }
  }
}
```

```typescript
class ShooterScene extends Phaser.Scene {
  private gameObjectRenderer: GameObjectRenderer
  private activeProjectiles: string[] = []
  
  create() {
    this.gameObjectRenderer = new GameObjectRenderer(this)
    
    this.gameObjectRenderer.init({
      enableObjectPool: true,
      poolSize: 500,  // 大量子弹
      enableAnimation: true
    })
    
    // 创建玩家飞机
    this.gameObjectRenderer.createObject({
      id: 'player',
      type: 'player',
      x: 400,
      y: 500,
      width: 48,
      height: 48,
      textureKey: 'player_ship',
      depth: 50,
      rotation: 0
    })
    
    // 监听射击事件
    EventBus.getInstance().on(
      GameEventType.INPUT_DIRECTION_CHANGED,
      (event) => {
        if (event.payload.action === 'shoot') {
          this.fireProjectile()
        }
      }
    )
  }
  
  update(deltaTime: number) {
    this.gameObjectRenderer.update(deltaTime)
    
    // 更新所有子弹位置
    this.activeProjectiles.forEach(id => {
      const config = this.gameObjectRenderer.getConfig(id)
      if (config) {
        const newPos = { x: config.x, y: config.y - 10 }  // 向上飞行
        this.gameObjectRenderer.updatePosition(id, newPos, false)
        
        // 移除超出屏幕的子弹
        if (newPos.y < 0) {
          this.gameObjectRenderer.destroyObject(id)
          this.activeProjectiles = this.activeProjectiles.filter(pid => pid !== id)
        }
      }
    })
  }
  
  private fireProjectile(): void {
    const playerConfig = this.gameObjectRenderer.getConfig('player')
    if (!playerConfig) return
    
    const id = `projectile_${Date.now()}`
    
    this.gameObjectRenderer.createObject({
      id,
      type: 'projectile',
      x: playerConfig.x,
      y: playerConfig.y - 30,
      width: 8,
      height: 20,
      color: 0xffff00,
      depth: 20
    })
    
    this.activeProjectiles.push(id)
    
    // 播放发射音效
    playShootSound()
  }
}
```

### 1. 对象池模式

```typescript
// ⭐ 避免频繁的内存分配和垃圾回收
class GameObjectRenderer {
  private objectPools: Map<GameObjectType, Group> = new Map()
  
  // 获取对象时优先从池中取
  createObject(config): GameObject {
    if (enableObjectPool) {
      const pool = objectPools.get(config.type)
      const pooledObject = pool.getFirstDead(false)
      
      if (pooledObject) {
        // 复用！无需新分配内存
        return pooledObject
      }
    }
    
    // 池中无可用对象时才创建新的
    return createGameObject(config)
  }
  
  // 销毁时回收到池中
  destroyObject(id): void {
    if (enableObjectPool) {
      const pool = objectPools.get(config.type)
      
      if (pool.contains(gameObject)) {
        // 回收到池中，下次复用
        gameObject.setActive(false)
        gameObject.setVisible(false)
        return
      }
    }
    
    // 真正销毁
    gameObject.destroy()
  }
}
```

### 2. 多态渲染策略

```typescript
createGameObject(config): GameObject {
  // 策略 1: 优先使用纹理（Sprite/Image）
  if (config.textureKey) {
    if (config.width && config.height) {
      gameObject = scene.add.image(x, y, textureKey)
      image.setDisplaySize(width, height)
    } else if (config.radius) {
      gameObject = scene.add.sprite(x, y, textureKey)
      sprite.setScale(radius / 32)
    }
  }
  // 策略 2: 使用图形（Graphics）
  else if (config.color !== undefined) {
    graphics = scene.add.graphics()
    graphics.fillStyle(color, alpha)
    
    if (config.radius) {
      graphics.fillCircle(x, y, radius)
    } else if (config.width && config.height) {
      graphics.fillRect(x - width/2, y - height/2, width, height)
    }
    
    gameObject = graphics
  }
  // 策略 3: 默认矩形
  else {
    graphics = scene.add.graphics()
    graphics.fillStyle(0xffffff, alpha)
    graphics.fillRect(x - width/2, y - height/2, width, height)
    gameObject = graphics
  }
  
  return gameObject
}
```
