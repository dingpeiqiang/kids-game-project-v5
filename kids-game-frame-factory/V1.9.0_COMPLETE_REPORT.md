```typescript
// ⭐ 高亮单个单元格
highlightCell(gridPos: GridPosition, color?: number, alpha: number = 0.5): void {
  // 清除之前的高亮
  clearHighlight()
  
  // 保存当前高亮位置
  currentHighlight = { ...gridPos }
  
  // 创建高亮图形
  if (!highlightGraphics) {
    highlightGraphics = scene.add.graphics()
    highlightGraphics.setDepth(-998)
  }
  
  const x = gridPos.col * cellSize
  const y = gridPos.row * cellSize
  
  highlightGraphics.fillStyle(color ?? defaultHighlightColor, alpha)
  highlightGraphics.fillRect(x, y, cellSize, cellSize)
}

// ⭐ 清除高亮
clearHighlight(): void {
  if (highlightGraphics) {
    highlightGraphics.clear()
  }
  currentHighlight = null
}
```

#### 3. 单元格状态管理 ⭐⭐⭐⭐⭐

```typescript
interface CellState {
  gridPos: GridPosition      // 网格坐标
  highlighted: boolean       // 是否高亮
  highlightColor?: number    // 高亮颜色
  disabled: boolean          // 是否禁用
  customColor?: number       // 自定义颜色
  alpha: number              // 透明度
}

// ⭐ 状态映射表
cellStates: Map<string, CellState> = new Map()

// ⭐ 更新单元格状态
updateCellState(gridPos, updates): void {
  const key = getCellKey(gridPos)  // "col_row"
  let state = cellStates.get(key)
  
  if (!state) {
    state = {
      gridPos,
      highlighted: false,
      disabled: false,
      alpha: 1
    }
    cellStates.set(key, state)
  }
  
  Object.assign(state, updates)
}

// ⭐ 获取单元格状态
getCellState(gridPos): CellState | undefined {
  const key = getCellKey(gridPos)
  return cellStates.get(key)
}
```

**使用示例**:
```typescript
// 设置单元格颜色
gridRenderer.setCellColor({ col: 5, row: 3 }, 0xff0000)

// 禁用单元格
gridRenderer.disableCell({ col: 10, row: 5 })

// 启用单元格
gridRenderer.enableCell({ col: 10, row: 5 })

// 获取状态
const state = gridRenderer.getCellState({ col: 5, row: 3 })
console.log(state?.highlighted)  // true/false
```

#### 4. 网格渲染算法 ⭐⭐⭐⭐⭐

```typescript
renderGrid(): void {
  const width = cols * cellSize
  const height = rows * cellSize
  
  // 1. 清除旧图形
  clearAll()
  
  // 2. 创建背景
  gridGraphics = scene.add.graphics()
  gridGraphics.fillStyle(backgroundColor, backgroundAlpha)
  gridGraphics.fillRect(0, 0, width, height)
  gridGraphics.setDepth(-999)
  
  // 3. 绘制网格线
  drawGridLines(width, height)
  
  // 4. 显示坐标（可选）
  if (showCoordinates) {
    drawCoordinates()
  }
}

drawGridLines(width, height): void {
  // 绘制垂直线
  for (let x = 0; x <= width; x += cellSize) {
    lineBetween(x, 0, x, height)
  }
  
  // 绘制水平线
  for (let y = 0; y <= height; y += cellSize) {
    lineBetween(0, y, width, y)
  }
}
```

**效果**:
```
┌───┬───┬───┬───┐
│   │   │   │   │  高度 = rows × cellSize
├───┼───┼───┼───┤
│   │   │   │   │
└───┴───┴───┴───┘
  宽度 = cols × cellSize
```

```typescript
  create() {
    this.gridRenderer = new GridRenderer(this)
    
    // 初始化网格（与游戏区域一致）
    this.gridRenderer.init({

        cols: 32,
        rows: 18,
        cellSize: 40,
        lineColor: 0x333333,
        lineWidth: 1,
        backgroundColor: 0x1a1a1a,
        backgroundAlpha: 0.9
      },
      enableHighlight: true,
      enableAnimation: true
    })
    
    // 监听游戏事件
    EventBus.getInstance().on(
      GameEventType.GAME_START,
      () => this.gridRenderer.clearHighlight()
    )
    
    EventBus.getInstance().on(
      GameEventType.PAUSE,
      () => {
        // 暂停时变暗网格
        if (this.gridRenderer) {
          this.scene.tweens.add({
            targets: this.gridRenderer['gridGraphics'],
            alpha: 0.3,
            duration: 300
          })
        }
      }
    )
  }
  
  update(deltaTime: number) {
    this.gridRenderer.update(deltaTime)
  }
}
```

```typescript
class MinesweeperScene extends Phaser.Scene {
  private gridRenderer: GridRenderer
  private revealedCells: Set<string> = new Set()
  private flaggedCells: Set<string> = new Set()
  
  create() {
    this.gridRenderer = new GridRenderer(this)
    
    this.gridRenderer.init({
      grid: {
        cols: 10,
        rows: 10,
        cellSize: 40,
        lineColor: 0x888888,
        lineWidth: 1,
        backgroundColor: 0xcccccc,
        highlightColor: 0x00ff00
      }
    })
    
    // 初始化地雷
    this.placeMines()
    
    // 点击事件
    this.input.on('pointerdown', (pointer) => {
      const col = Math.floor(pointer.x / 40)
      const row = Math.floor(pointer.y / 40)
      
      if (pointer.rightButtonDown()) {
        this.toggleFlag({ col, row })
      } else {
        this.revealCell({ col, row })
      }
    })
  }
  
  private revealCell(pos: GridPosition): void {
    const key = `${pos.col}_${pos.row}`
    
    if (this.revealedCells.has(key)) return
    
    this.revealedCells.add(key)
    
    // 重置单元格颜色（表示已揭开）
    this.gridRenderer.resetCellColor(pos)
    
    // 如果是地雷，游戏结束
    if (this.isMine(pos)) {
      this.gameOver()
    }
  }
  
  private toggleFlag(pos: GridPosition): void {
    const key = `${pos.col}_${pos.row}`
    
    if (this.flaggedCells.has(key)) {
      this.flaggedCells.delete(key)
      this.gridRenderer.enableCell(pos)
    } else {
      this.flaggedCells.add(key)
      this.gridRenderer.disableCell(pos)  // 标记为禁用状态
    }
  }
}
```
