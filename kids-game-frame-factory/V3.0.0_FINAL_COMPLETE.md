public removeComponent(key: string): void {
  const component = this.components[key]
  
  if (component) {
    // 自动销毁
    if ('destroy' in component) {
      component.destroy()
    }
    
    delete this.components[key]
    console.log(`🗑️ 移除组件：${key}`)
  }
}

public getComponent<T extends IComponent>(key: string): T | undefined {

### 完整使用示例

#### 贪吃蛇游戏场景

```typescript
import { 
  ComponentGameScene,
  BackgroundRenderer,
  GridRenderer,
  GameObjectRenderer,
  ParticleRenderer,
  GameStateComponent,
  InputHandlerComponent,
  BackgroundType
} from 'kids-game-frame-factory'

class SnakeGameScene extends ComponentGameScene {
  private snakeBody: string[] = []
  
  create() {
    super.create()
    
    // 1. 添加背景渲染器
    this.components.background = new BackgroundRenderer(this)
    this.components.background.init({
      background: {
        type: BackgroundType.GRADIENT,
        gradientStart: 0x1a237e,
        gradientEnd: 0x0d47a1
      }
    })
    
    // 2. 添加网格渲染器
    this.components.grid = new GridRenderer(this)
    this.components.grid.init({
      grid: {
        cellSize: 20,
        columns: 20,
        rows: 20,
        color: 0xffffff,
        showGrid: true
      }
    })
    
    // 3. 添加游戏对象渲染器
    this.components.gameObject = new GameObjectRenderer(this)
    this.components.gameObject.init({
      enableObjectPool: true,
      poolSize: 100,
      enableAnimation: true
    })
    
    // 4. 添加粒子渲染器
    this.components.particle = new ParticleRenderer(this)
    this.components.particle.init({
      defaultCount: 30,
      maxActiveEffects: 50
    })
    
    // 5. 添加游戏状态管理器
    this.components.gameState = new GameStateComponent(this)
    this.components.gameState.init({
      initialState: 'playing'
    })
    
    // 6. 添加输入处理器
    this.components.input = new InputHandlerComponent(this)
    this.components.input.init({
      enableKeyboard: true,
      enableTouch: true,
      preventDefaultKeys: ['ArrowUp', 'ArrowDown', 'ArrowLeft', 'ArrowRight']
    })
    
    // 7. 创建初始蛇
    this.createSnake()
    
    // 8. 启动游戏
    this.startGame()
    
    console.log('🐍 贪吃蛇游戏已启动')
  }
  
  update(_time: number, delta: number) {
    super.update(_time, delta)
    
    // 添加自定义更新逻辑
    if (this.components.gameState?.getState() === 'playing') {
      this.moveSnake(delta)
      this.checkCollisions()
    }
  }
  
  private createSnake(): void {
    const gameObject = this.components.gameObject as GameObjectRenderer
    
    // 创建蛇头
    gameObject.createObject({
      id: 'snake_head',
      type: 'snake',
      x: 200,
      y: 200,
      width: 18,
      height: 18,
      color: 0x00ff00,
      depth: 30
    })
    
    this.snakeBody.push('snake_head')
    
    // 创建蛇身
    for (let i = 1; i < 3; i++) {
      gameObject.createObject({
        id: `snake_body_${i}`,
        type: 'snake',
        x: 200 - i * 20,
        y: 200,
        width: 18,
        height: 18,
        color: 0x00cc00,
        depth: 30
      })
      
      this.snakeBody.push(`snake_body_${i}`)
    }
  }
  
  private moveSnake(delta: number): void {
    // 获取当前方向
    const input = this.components.input as InputHandlerComponent
    const direction = input.getCurrentDirection()
    
    // 移动蛇头
    const head = this.components.gameObject?.getObject('snake_head')
    if (head) {
      const newX = head.x + direction.x * 0.2 * delta
      const newY = head.y + direction.y * 0.2 * delta
      
      this.components.gameObject?.updatePosition('snake_head', { x: newX, y: newY }, true)
    }
    
    // 蛇身跟随...
  }
  
  private checkCollisions(): void {
    // 检测碰撞
    const collisionDetection = this.components.collision as CollisionDetectionComponent
    
    if (collisionDetection) {
      // 检测撞墙
      if (this.detectWallCollision()) {
        // 播放爆炸特效
        const particle = this.components.particle as ParticleRenderer
        particle?.playPresetEffect({
          type: 'explosion',
          position: { x: 200, y: 200 }
        })
        
        // 游戏结束
        this.endGame('collision_wall')
      }
      
      // 检测吃食物...
    }
  }
  
  private detectWallCollision(): boolean {
    const head = this.components.gameObject?.getObject('snake_head')
    if (!head) return false
    
    const grid = this.components.grid as GridRenderer
    const config = grid.getConfig()
    
    if (!config) return false
    
    return (
      head.x < 0 ||
      head.x > config.columns * config.cellSize ||
      head.y < 0 ||
      head.y > config.rows * config.cellSize
    )
  }
}

// 使用场景
const game = new Phaser.Game({
  type: Phaser.AUTO,
  width: 800,
  height: 600,
  scene: [SnakeGameScene],
  parent: 'game-container'
})
```

      id: 'player',
      type: 'player',
      x: 100,
      y: 400,
      width: 32,
      height: 48,
      textureKey: 'player_sprite',
      depth: 50
    })
  }

    const platforms = [
      { x: 0, y: 550, width: 800, height: 50 },
      { x: 200, y: 400, width: 200, height: 20 },
      { x: 500, y: 300, width: 200, height: 20 }
    ]
    
    platforms.forEach((platform, index) => {
      this.components.gameObject?.createObject({
        id: `platform_${index}`,
        type: 'obstacle',
        x: platform.x,
        y: platform.y,
        width: platform.width,
        height: platform.height,
        color: 0x8b4513,
        depth: -100
      })
    })
  }
  
  private createEnemies(): void {
    for (let i = 0; i < 5; i++) {
      this.components.gameObject?.createObject({
        id: `enemy_${i}`,
        type: 'enemy',
        x: 200 + i * 150,
        y: 200,
        width: 32,
        height: 32,
        textureKey: 'enemy_sprite',
        depth: 30
      })
    }
  }
}

      textureKey: 'player_sprite'
    })
    
    // 3. 启动游戏
    this.startGame()
  }
  
  update(time: number, delta: number) {
    super.update(time, delta)
    
    // 添加自定义逻辑
    if (this.components.gameState.getState() === 'playing') {
      this.handleInput()
      this.updatePhysics()
      this.checkCollisions()
    }
  }
}
