### 核心功能

#### 1. 多类型碰撞检测 ⭐⭐⭐⭐⭐

```typescript
// ✅ 矩形碰撞（AABB 算法）
detectRectangleCollision(posA, widthA, heightA, posB, widthB, heightB)

// ✅ 圆形碰撞（距离检测）
detectCircleCollision(centerA, radiusA, centerB, radiusB)

// ✅ 点碰撞（点在矩形内检测）
isPointInRectangle(point, rectangle)
```

**使用示例**:
```typescript
const collision = new CollisionDetectionComponent(scene)

// 注册矩形碰撞体
collision.registerCollider(player, {
  type: ColliderType.RECTANGLE,
  position: { x: 100, y: 100 },
  width: 50,
  height: 50,
  enabled: true
})

// 注册圆形碰撞体
collision.registerCollider(enemy, {
  type: ColliderType.CIRCLE,
  position: { x: 200, y: 150 },
  radius: 30,
  enabled: true
})

// 添加碰撞对

```

#### 2. 碰撞事件系统 ⭐⭐⭐⭐⭐

```typescript
// ✅ 碰撞开始（首次接触）
COLLISION_DETECTED (eventType: 'start')

// ✅ 碰撞持续（每帧检测，可选）
COLLISION_DETECTED (eventType: 'stay')

// ✅ 碰撞结束（分离）
COLLISION_DETECTED (eventType: 'end')
```

**事件监听示例**:
```typescript
EventBus.getInstance().on(
  GameEventType.COLLISION_DETECTED,
  (event) => {
    const { eventType, objectA, objectB } = event.payload
    
    switch (eventType) {
      case 'start':
        console.log(`💥 碰撞开始：${objectA} ↔ ${objectB}`)
        // 播放碰撞音效、扣血等
        break
        
      case 'stay':
        console.log(`🔄 持续碰撞：${objectA} ↔ ${objectB}`)
        // 持续伤害效果
        break
        
      case 'end':
        console.log(`✅ 碰撞结束：${objectA} ↔ ${objectB}`)
        // 清理状态
        break
    }
  }
)
```

### 3. 碰撞状态机

```typescript
// ⭐ 记录上一次的碰撞状态
previousCollisions: Set<string>
currentCollisions: Set<string>

// ⭐ 判断碰撞阶段
if (!previous.has(id) && current.has(id)) {
  emitCollisionStart()  // 新碰撞
} else if (previous.has(id) && !current.has(id)) {
  emitCollisionEnd()    // 碰撞结束
} else if (current.has(id)) {
  emitCollisionStay()   // 持续碰撞
}
```

## 🚀 可以立即使用

### 完整的游戏组装示例

```typescript
import { 
  GameStateComponent,
  GridMovementComponent,
  ScoreManagerComponent,
  InputHandlerComponent,
  CollisionDetectionComponent
} from 'kids-game-frame-factory'

class SnakeGameScene extends Phaser.Scene {
  create() {
    // 创建所有组件
    const gameState = new GameStateComponent(this)
    const gridMovement = new GridMovementComponent(this)
    const scoreManager = new ScoreManagerComponent(this)
    const inputHandler = new InputHandlerComponent(this)
    const collision = new CollisionDetectionComponent(this)
    
    // 初始化
    gameState.init({})
    gridMovement.init({ speed: 200, gridCols: 32, gridRows: 18, cellSize: 40 })
    scoreManager.init({ enableCombo: true })
    inputHandler.init({ enableKeyboard: true })
    collision.init({ debugMode: false })
    
    // 注册碰撞体
    collision.registerCollider(snakeHead, snakeHeadCollider)
    collision.registerCollider(food, foodCollider)
    collision.addCollisionPair(snakeHead, food)
    
    // 监听碰撞事件
    EventBus.getInstance().on(
      GameEventType.COLLISION_DETECTED,
      (event) => {
        if (event.payload.objectB === 'food') {
          scoreManager.addScore(10)
          spawnNewFood()
        }
      }
    )
    
    // 开始游戏
    gameState.startGame()
  }
  
  update(time: number, delta: number) {
    // 更新所有组件
    gridMovement.update(delta)
    scoreManager.update(delta)
    inputHandler.update(delta)
    collision.update(delta)
  }
}
```
