```typescript
enum DifficultyLevel {
```

```typescript
// ⭐ 自动保存配置
saveConfig(): void {
  const serialized = JSON.stringify(this.config)
  localStorage.setItem(storageKey, serialized)
}

// ⭐ 加载保存的配置
loadConfig(): boolean {
  const serialized = localStorage.getItem(storageKey)
  if (serialized) {
    this.config = JSON.parse(serialized)
    return true
  }
  return false
}

// ⭐ 运行时自动保存
setConfigItem(key, value): void {
  this.config[key] = value
  this.isDirty = true
  
  if (autoSave) {
    this.saveConfig()  // 立即保存
  }
}
```

**效果**:
- 玩家调整的难度、音量等设置会自动保存
- 下次打开游戏时自动加载上次的配置
- 无需手动保存，用户体验极佳

```typescript
setConfigItem(key, value, emitEvent = true): void {
  const oldValue = this.config[key]

    this.emit({
```

```typescript
import { GameConfigManager, DifficultyLevel } from 'kids-game-frame-factory'

class SettingsScene extends Phaser.Scene {
  private configManager: GameConfigManager
  
  create() {
    this.configManager = new GameConfigManager(this)
    this.configManager.init({
      defaultDifficulty: DifficultyLevel.NORMAL,
      autoSave: true
    })
    
    // 创建设置 UI
    this.createSettingsUI()
  }
  
  private createSettingsUI(): void {
    const width = this.scale.width
    let y = 100
    
    // 难度选择
    this.add.text(width / 2, y, '难度选择', {
      fontSize: '24px',
      color: '#ffffff'
    })
    
    const difficulties = ['easy', 'normal', 'hard', 'expert']
    const currentDiff = this.configManager.getDifficulty()
    
    difficulties.forEach((diff, index) => {
      const btn = this.add.text(width / 2, y + 50 + index * 60, diff.toUpperCase(), {
        fontSize: '20px',
        color: diff === currentDiff ? '#4CAF50' : '#ffffff',
        backgroundColor: diff === currentDiff ? '#2E7D32' : '#333333',
        padding: { x: 20, y: 10 }
      })
      
      btn.setOrigin(0.5)
      btn.setInteractive({ useHandCursor: true })
      btn.on('pointerdown', () => {
        this.configManager.setDifficulty(diff as DifficultyLevel)
        this.scene.restart()  // 刷新 UI
      })
    })
    
    // 音量滑块
    y += 400
    this.add.text(width / 2, y, '主音量', {
      fontSize: '24px',
      color: '#ffffff'
    })
    
    const volumeText = this.add.text(width / 2, y + 50, 
      `${Math.round(this.configManager.getConfigItem('masterVolume') * 100)}%`, {
        fontSize: '20px',
        color: '#ffffff'
      })
    volumeText.setOrigin(0.5)
    
    // 点击调整音量
    volumeText.setInteractive({ useHandCursor: true })
    volumeText.on('pointerdown', () => {
      const current = this.configManager.getConfigItem('masterVolume')
      const newVolume = (current >= 0.9 ? 0.1 : current + 0.1)
      this.configManager.setConfigItem('masterVolume', newVolume)
      volumeText.setText(`${Math.round(newVolume * 100)}%`)
    })
    
    // 粒子效果开关
    y += 150
    const particlesEnabled = this.configManager.getConfigItem('enableParticles')
    const particlesText = this.add.text(width / 2, y, 
      `粒子效果：${particlesEnabled ? '开启' : '关闭'}`, {
        fontSize: '20px',
        color: '#ffffff'
      })
    particlesText.setOrigin(0.5)
    particlesText.setInteractive({ useHandCursor: true })
    particlesText.on('pointerdown', () => {
      const current = this.configManager.getConfigItem('enableParticles')
      this.configManager.setConfigItem('enableParticles', !current)
      particlesText.setText(`粒子效果：${!current ? '开启' : '关闭'}`)
    })
  }
}
```

#### 动态难度调整（DDA）

```typescript
class SmartGameScene extends Phaser.Scene {
  private configManager: GameConfigManager
  private playerPerformance: number = 0  // 玩家表现分数
  
  create() {
    this.configManager = new GameConfigManager(this)
    this.configManager.init({
      defaultDifficulty: DifficultyLevel.NORMAL,
      autoSave: true
    })
  }
  
  update(deltaTime: number) {
    this.configManager.update(deltaTime)
    
    // 根据玩家表现动态调整难度
    this.checkAndAdjustDifficulty()
  }
  
  private checkAndAdjustDifficulty(): void {
    // 玩家连续得分 → 提高难度
    if (this.playerPerformance > 1000) {
      const current = this.configManager.getDifficulty()
      if (current === DifficultyLevel.NORMAL) {
        this.configManager.setDifficulty(DifficultyLevel.HARD)
        this.showMessage('难度提升！准备好迎接挑战了吗？')
      }
    }
    
    // 玩家多次失败 → 降低难度
    if (this.playerPerformance < -500) {
      const current = this.configManager.getDifficulty()
      if (current === DifficultyLevel.HARD) {
        this.configManager.setDifficulty(DifficultyLevel.NORMAL)
        this.showMessage('难度已降低，继续加油！')
      }
    }
  }
  
  private showMessage(text: string): void {
    // 显示提示消息
  }
}
```

```typescript
class BootScene extends Phaser.Scene {
  private configManager: GameConfigManager
  
  preload() {
    this.configManager = new GameConfigManager(this)
    this.configManager.init({
      defaultDifficulty: DifficultyLevel.NORMAL,
      autoSave: true,
      useLocalStorage: true
    })
    
    // 加载玩家保存的配置
    const loaded = this.configManager.loadConfig()
    
    if (loaded) {
      console.log('✅ 已加载玩家配置')
      
      // 应用配置
      const config = this.configManager.getConfig()
      
      // 设置音量
      this.sound.volume = config.masterVolume
      
      // 设置特效
      if (!config.enableParticles) {
        this.disableParticleEffects()
      }
      if (!config.enableScreenShake) {
        this.disableScreenShake()
      }
    } else {
      console.log('ℹ️ 首次启动，使用默认配置')
    }
  }
  
  create() {
    // 进入主菜单
    this.scene.start('MainMenu')
  }
}
```

### 完整的游戏配置流程

```typescript
import { 
  GameConfigManager,
  DifficultyLevel,
  GameStateComponent,
  ScoreManagerComponent
} from 'kids-game-frame-factory'

class CompleteGameScene extends Phaser.Scene {
  private configManager: GameConfigManager
  private gameState: GameStateComponent
  private scoreManager: ScoreManagerComponent
  
  create() {
    // 创建配置管理器
    this.configManager = new GameConfigManager(this)
    this.configManager.init({
      defaultDifficulty: DifficultyLevel.NORMAL,
      autoSave: true
    })
    
    // 创建其他组件
    this.gameState = new GameStateComponent(this)
    this.scoreManager = new ScoreManagerComponent(this)
    
    // 获取难度配置
    const diffConfig = this.configManager.getCurrentDifficultyConfig()
    
    // 应用难度配置
    this.gameState.init({
      lives: diffConfig.lives
    })
    
    this.scoreManager.init({
      enableCombo: this.configManager.getConfigItem('enableCombo'),
      comboDuration: diffConfig.comboThreshold
    })
    
    // 监听难度变更
    EventBus.getInstance().on(
      GameEventType.SCORE_CHANGED,
      (event) => {
        if (event.payload.difficulty) {
          const newDiff = event.payload.difficulty
          console.log(`难度已调整为：${newDiff}`)
          this.applyNewDifficulty(newDiff)
        }
      }
    )
    
    // 开始游戏
    this.gameState.startGame()
  }
  
  private applyNewDifficulty(difficulty: DifficultyLevel): void {
    const diffConfig = this.configManager.getDifficultyConfig(difficulty)
    
    // 调整游戏速度
    this.time.timeScale = diffConfig.speedMultiplier
    
    // 调整得分倍率
    this.scoreManager.setMultiplier(diffConfig.scoreMultiplier)
    
    // 显示提示
    this.showMessage(`难度已调整为：${difficulty}`)
  }
}
```
