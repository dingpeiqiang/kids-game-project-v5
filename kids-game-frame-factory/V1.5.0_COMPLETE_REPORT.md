```typescript
type ItemType = 'normal' | 'bonus' | 'special' | 'powerup'

interface GameItem {
```

```typescript
// ✅ 避免与障碍物重叠
overlapsWithObstacles(position, obstacles)

overlapsWithItems(position)

// ✅ 排除指定位置
isInExcludeList(position, excludePositions)

// ✅ 最大尝试次数（防止死循环）
MAX_SPAWN_ATTEMPTS = 100
```

**算法流程**:
```typescript
spawnItem(obstacles, excludePositions): GameItem {
  for (attempt = 0; attempt < 100; attempt++) {
    position = randomPosition()
    
    if (overlapsWithObstacles(position)) continue
    if (isInExcludeList(position)) continue
    if (overlapsWithItems(position)) continue
    
    // 找到有效位置
    return createItem(position)
  }
  
  // 无法找到位置，返回虚拟物品
  return createVirtualItem()
}
```

#### 3. 物品生命周期管理 ⭐⭐⭐⭐⭐

```typescript
// ✅ 每帧更新检查过期物品
update(deltaTime): void {
  const now = Date.now()
  
  items.forEach(item => {
    const age = now - item.spawnedAt
    if (age > item.lifetime && !item.collected) {
      item.collected = true
      emitItemExpired(item)
    }
  })
  
  // 移除已收集的物品
  items = items.filter(i => !i.collected)
}
```

**使用场景**:
```typescript
// 限时物品（10 秒后消失）
spawner.init({ itemLifetime: 10000 })

// 永久物品（不设置 lifetime）
spawner.init({})

// 定时清理
EventBus.on(ITEM_EFFECT_EXPIRED, (event) => {
  console.log(`⏰ 物品 ${event.itemId} 已过期`)
  removeGameObject(event.itemId)
})
```

```typescript
import { ItemSpawnerComponent, ScoreManagerComponent } from 'kids-game-frame-factory'

class SnakeGameScene extends Phaser.Scene {
  private itemSpawner: ItemSpawnerComponent
  private scoreManager: ScoreManagerComponent
  
  create() {
    // 创建组件
    this.itemSpawner = new ItemSpawnerComponent(this)
    this.scoreManager = new ScoreManagerComponent(this)
    
    // 初始化
    this.itemSpawner.init({
      availableTypes: ['normal', 'bonus'],
      gridCols: 32,
      gridRows: 18,
      cellSize: 40,
      typeProbabilities: {
        normal: 0.8,
        bonus: 0.2
      },
      itemPoints: {
        normal: 10,
        bonus: 50
      },
      maxItems: 1  // 只允许同时存在 1 个食物
    })
    
    this.scoreManager.init({ enableCombo: true })
    
    // 生成第一个食物
    const food = this.itemSpawner.spawnItem(snakeBody)
    
    // 监听物品收集事件
    EventBus.getInstance().on(
      GameEventType.ITEM_COLLECTED,
      (event) => {
        const { itemId, itemType, points } = event.payload
        
        // 加分
        this.scoreManager.addScore(points)
        
        // 播放音效
        playCollectSound()
        
        // 生成新食物
        setTimeout(() => {
          this.itemSpawner.spawnItem(snakeBody)
        }, 100)
      }
    )
  }
  
  update(time: number, delta: number) {
    this.itemSpawner.update(delta)
    this.scoreManager.update(delta)
  }
}
```

```typescript
class PlatformerScene extends Phaser.Scene {
  private coinSpawner: ItemSpawnerComponent

      availableTypes: ['normal', 'bonus', 'powerup'],
      gridCols: 50,
      gridRows: 20,
      cellSize: 32,
      maxItems: 10,

        normal: 1,
        bonus: 5,
        powerup: 20
      }
    })
    
    // 在关卡中生成金币
    const platforms = [
      { x: 100, y: 300, width: 200, height: 20 },
      { x: 400, y: 200, width: 150, height: 20 }
    ]
    
    // 在每个平台上生成 3-5 个金币
    platforms.forEach(platform => {
      const count = randomInt(3, 5)
      for (let i = 0; i < count; i++) {
        this.coinSpawner.spawnItem([], [{
          x: platform.x + i * 40,
          y: platform.y - 40,
          width: 32,
          height: 32
        }])
      }
    })
    
    // 监听收集事件
    EventBus.getInstance().on(
      GameEventType.ITEM_COLLECTED,
      (event) => {
        addCoinToUI(event.points)
        playCoinSound()
      }
    )
  }
}
```

```typescript
        powerup: 500
      }
    })

    EventBus.getInstance().on(
      GameEventType.ENEMY_DESTROYED,
      (event) => {
        if (Math.random() < 0.3) {  // 30% 概率
          const obstacle = getEnemyBounds(event.enemyId)
          this.powerupSpawner.spawnItem([obstacle])
        }
      }
    )
    
    // 监听道具收集
    EventBus.getInstance().on(
      GameEventType.ITEM_COLLECTED,
      (event) => {
        switch (event.itemType) {
          case 'normal':
            upgradeWeapon()
            break
          case 'bonus':
            restoreHealth()
            break
          case 'special':
            activateShield()
            break
          case 'powerup':
            activateBomb()
            break
        }
        this.scoreManager.addScore(event.points)
      }
    )
  }
}
```

## 🚀 可以立即使用

### 完整的贪吃蛇游戏组装

```typescript
import { 
  GameStateComponent,
  GridMovementComponent,
  ScoreManagerComponent,
  InputHandlerComponent,
  CollisionDetectionComponent,
  ItemSpawnerComponent
} from 'kids-game-frame-factory'

class SnakeGameScene extends Phaser.Scene {
  create() {
    // 创建所有组件
    const gameState = new GameStateComponent(this)
    const gridMovement = new GridMovementComponent(this)
    const scoreManager = new ScoreManagerComponent(this)
    const inputHandler = new InputHandlerComponent(this)
    const collision = new CollisionDetectionComponent(this)
    const itemSpawner = new ItemSpawnerComponent(this)
    
    // 初始化
    gameState.init({})
    gridMovement.init({ speed: 200, gridCols: 32, gridRows: 18, cellSize: 40 })
    scoreManager.init({ enableCombo: true })
    inputHandler.init({ enableKeyboard: true })
    collision.init({ debugMode: false })
    itemSpawner.init({
      availableTypes: ['normal', 'bonus'],
      gridCols: 32,
      gridRows: 18,
      cellSize: 40,
      maxItems: 1
    })
    
    // 注册碰撞体
    collision.registerCollider(snakeHead, snakeHeadCollider)
    collision.registerCollider(food, foodCollider)
    collision.addCollisionPair(snakeHead, food)
    
    // 生成第一个食物
    itemSpawner.spawnItem(snakeBody)
    
    // 监听碰撞事件
    EventBus.getInstance().on(
      GameEventType.COLLISION_DETECTED,
      (event) => {
        if (event.payload.objectB === 'food') {
          itemSpawner.collectItem(event.payload.itemId)
          scoreManager.addScore(event.payload.points)
          itemSpawner.spawnItem(snakeBody)  // 生成新食物
        }
      }
    )
    
    // 开始游戏
    gameState.startGame()
  }
  
  update(time: number, delta: number) {
    // 更新所有组件
    gridMovement.update(delta)
    scoreManager.update(delta)
    inputHandler.update(delta)
    collision.update(delta)
    itemSpawner.update(delta)
  }
}
```
