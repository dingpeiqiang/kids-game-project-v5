```typescript
renderGradient(config): void {
  const width = scene.scale.width
  const height = scene.scale.height
  
  // 创建渐变纹理
  const graphics = scene.make.graphics({ x: 0, y: 0 })
  
  const steps = direction === 'vertical' ? height : width
  
  for (let i = 0; i < steps; i++) {
    const t = i / steps  // 插值因子
    
    // RGB 颜色插值
    const r = Math.round(R1 * (1-t) + R2 * t)
    const g = Math.round(G1 * (1-t) + G2 * t)
    const b = Math.round(B1 * (1-t) + B2 * t)
    const color = (r << 16) | (g << 8) | b
    
    // 绘制一条线
    if (direction === 'vertical') {
      graphics.fillStyle(color, alpha)
      graphics.fillRect(0, i, width, 1)
    } else {
      graphics.fillStyle(color, alpha)
      graphics.fillRect(i, 0, 1, height)
    }
  }
  
  // 生成纹理并创建 Sprite
  graphics.generateTexture('gradient_bg', width, height)
  graphics.destroy()
}
```

#### 3. 视差滚动效果 ⭐⭐⭐⭐⭐

```typescript
update(deltaTime): void {
  // 处理平铺背景的视差滚动
  if (enableParallax && tileSprite) {
    const seconds = deltaTime / 1000
    
    if (scrollSpeedX !== undefined) {
      tileSprite.tilePositionX += scrollSpeedX * seconds
    }
    
    if (scrollSpeedY !== undefined) {
      tileSprite.tilePositionY += scrollSpeedY * seconds
    }
  }
  
  // 处理普通背景的视差偏移
  if (enableParallax && backgroundSprite) {
    const factor = parallaxFactor ?? 0.5
    backgroundSprite.x = (camera.scrollX * factor) % width
    backgroundSprite.y = (camera.scrollY * factor) % height
  }
}
```

**效果说明**:
```
相机移动 → 背景以较慢速度移动 → 产生深度感

前景（玩家）：移动速度 100%
中景（背景）：移动速度 50% (parallaxFactor: 0.5)
远景（天空）：移动速度 20% (parallaxFactor: 0.2)
```

```typescript
renderGrid(config): void {
  const gridSize = config.gridSize ?? 40
  const gridColor = config.gridColor ?? 0x444444
  const gridLineWidth = config.gridLineWidth ?? 1
  
  // 绘制垂直线
  for (let x = 0; x <= width; x += gridSize) {
    graphics.lineStyle(gridLineWidth, gridColor, alpha)
    graphics.lineBetween(x, 0, x, height)
  }
  
  // 绘制水平线
  for (let y = 0; y <= height; y += gridSize) {
    graphics.lineStyle(gridLineWidth, gridColor, alpha)
    graphics.lineBetween(0, y, width, y)
  }
}
```

#### 动态背景切换（昼夜循环）

```typescript
class DayNightCycleScene extends Phaser.Scene {
  private bgRenderer: BackgroundRenderer
  private gameTime: number = 0  // 游戏时间（秒）
  
  create() {
    this.bgRenderer = new BackgroundRenderer(this)
    
    // 初始为白天
    this.bgRenderer.init({
      background: {
        type: BackgroundType.GRADIENT,
        gradientStart: 0x87CEEB,  // 天空蓝
        gradientEnd: 0xE0F6FF,    // 浅蓝
        gradientDirection: 'vertical'
      }
    })
  }
  
  update(deltaTime: number) {
    this.bgRenderer.update(deltaTime)
    
    // 每 60 秒一个昼夜循环
    this.gameTime += deltaTime / 1000
    
    const cycleProgress = (this.gameTime % 60) / 60  // 0-1
    
    if (cycleProgress < 0.25) {
      // 早晨（0-15 秒）
      this.updateDayBackground(cycleProgress * 4)
    } else if (cycleProgress < 0.5) {
      // 黄昏（15-30 秒）
      this.updateSunsetBackground((cycleProgress - 0.25) * 4)
    } else if (cycleProgress < 0.75) {
      // 夜晚（30-45 秒）
      this.updateNightBackground((cycleProgress - 0.5) * 4)
    } else {
      // 黎明（45-60 秒）
      this.updateDawnBackground((cycleProgress - 0.75) * 4)
    }
  }
  
  private updateDayBackground(progress: number): void {
    const startColor = this.lerpColor(0xFF6B6B, 0x87CEEB, progress)  // 橙红→天蓝
    const endColor = this.lerpColor(0xFFE66D, 0xE0F6FF, progress)    // 金黄→浅蓝
    
    this.bgRenderer.setBackground({
      type: BackgroundType.GRADIENT,
      gradientStart: startColor,
      gradientEnd: endColor,
      gradientDirection: 'vertical'
    })
  }
  
  private lerpColor(color1: number, color2: number, t: number): number {
    const r1 = (color1 >> 16) & 0xFF
    const g1 = (color1 >> 8) & 0xFF
    const b1 = color1 & 0xFF
    
    const r2 = (color2 >> 16) & 0xFF
    const g2 = (color2 >> 8) & 0xFF
    const b2 = color2 & 0xFF
    
    const r = Math.round(r1 * (1 - t) + r2 * t)
    const g = Math.round(g1 * (1 - t) + g2 * t)
    const b = Math.round(b1 * (1 - t) + b2 * t)
    
    return (r << 16) | (g << 8) | b
  }
}
```
