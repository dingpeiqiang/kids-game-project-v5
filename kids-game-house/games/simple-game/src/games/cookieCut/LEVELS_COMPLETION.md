```typescript
  rotationSpeed: number

```

### 3. 平滑过渡
```typescript
// 进入下一关时重置状态
function nextLevel() {
  currentLevel++
  levelConfig = getLevelConfig(currentLevel)
  state.gameStartTime = Date.now()
  state.gameEnded = false
  state.cookies = []
  state.combo = 0
  
  // 延迟2秒后开始，给玩家准备时间
  setTimeout(() => {
    state.cookies.push(spawnCookie(levelConfig))
  }, 500)
}
```

## 📝 使用说明

### 启动游戏
```typescript
// 从第1关开始（默认）
initCookieCut(engine, onEnd)

// 从指定关卡开始
initCookieCut(engine, onEnd, 5)  // 从第5关开始
```

### 查看关卡配置
```typescript
import { LEVELS, getLevelConfig } from './levels'

// 获取所有关卡
console.log(LEVELS)

// 获取特定关卡
const level5 = getLevelConfig(5)
```

### 调整关卡难度
编辑 `levels.ts` 中的对应关卡配置即可。
